00   read r1           # get # from user and store it in r1
01   setn r2 0         # set r2 as 0
02   write r2          # print the value of r2
03   mul r3 r1 r1      # r3 = r1 * r1
04   sub r4 r3 r2      # r4 = r3 - r2
05   jeqzn r4 08       # if r4 == 0, jump to line 08
06   addn r2 1         # add 1 to r2
07   jumpn 02          # jump to line 02
08   halt              # halt the program
